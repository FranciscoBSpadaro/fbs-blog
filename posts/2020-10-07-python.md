### Solução do problema:

```python

produtos = []
quantidade = []
valor = []
resultado = []


for cont in range(0,3):
    produtos.append(str(input('Digite o produto:')))
    quantidade.append(int(input('Digite a quantidade:')))
    valor.append(float(input('Digite o Valor , obs nao utilize virgula:')))
    
for x,y in enumerate(valor): # Percorre o Array a enumerando suas posições  x e y é só um parametro p valor e quant
     resultado.append(y*quantidade[x])  # adiciona o resultado da multiplicação de valor e quantidade na lista resultado  e y é valor * quantidade e [x] = é os valores dos resultados que vai nos índices da lista resultado 


print(f'O Produto:{produtos[0]} foi Vendido: {quantidade[0]} vezes ,  pelo valor de  R$ {valor[0]}  e o valor total foi de : R$ {resultado[0]} ')
print(f'O Produto:{produtos[1]} foi Vendido: {quantidade[1]} vezes ,  pelo valor de  R$ {valor[1]}  e o valor total foi de : R$ {resultado[1]} ')
print(f'O Produto:{produtos[2]} foi Vendido: {quantidade[2]} vezes ,  pelo valor de  R$ {valor[2]}  e o valor total foi de : R$ {resultado[2]} ')
print(f'O Valor Total de todas as Vendas é de  R$ {(sum(resultado))} ' )

# função sum soma todos os valores dos índices da lista resultado

```
